623  private Dictionary<Point, Point> BreadthFirstSearch(Func<Point, List<Point>> getNeighbours)
624  {
625  List<Point> frontiner = new List<Point>();
626  frontiner.Add(this.gameModel.MyPlayer.Cords);
627  Dictionary<Point, Point> path = new Dictionary<Point, Point>();
628  path.Add(this.gameModel.MyPlayer.Cords, new Point(0, 0)); // player's cord
629  while (frontiner.Count > 0)
630  {
631  Point current = frontiner[0];
632  frontiner.RemoveAt(0);
633  var neighbours = getNeighbours(current);
634  foreach (var next in neighbours)
635  {
636  if (!path.ContainsKey(next))
637  {
638  frontiner.Add(next);
639  path[next] = new Point(current.X - next.X, current.Y - next.Y); // generate direction vector to the next tile
640  }
641  }
642  }
643 
644  return path;
645  }

647  private List<Point> GetRegularEnemyNeighbours(Point current)
648  {
649  List<Point> neighbours = new List<Point>();
650  neighbours.Add(new Point(1, 0));
651  neighbours.Add(new Point(-1, 0));
652  neighbours.Add(new Point(0, 1));
653  neighbours.Add(new Point(0, -1));
654  List<Point> rmList = new List<Point>();
655  foreach (var item in neighbours)
656  {
657  // Basic tracking -> cant move through walls,water,goal
658  Point check = new Point(current.X + item.X, current.Y + item.Y);
659  if ((check.X < 0 || check.X >= (int)(this.gameModel.GameWidth / GameModel.TileSize)) || (check.Y < 0 || check.Y >= (int)(this.gameModel.GameHeight / GameModel.TileSize)))
660  {
661  rmList.Add(item);
662  }
663  else if (this.gameModel.GameAreaChar[(int)check.X, (int)check.Y] == 'W' ||
664  this.gameModel.GameAreaChar[(int)check.X, (int)check.Y] == 'P' ||
665  this.gameModel.GameAreaChar[(int)check.X, (int)check.Y] == 'G')
666  {
667  rmList.Add(item);
668  }
669  }
670 
671  foreach (var item in rmList)
